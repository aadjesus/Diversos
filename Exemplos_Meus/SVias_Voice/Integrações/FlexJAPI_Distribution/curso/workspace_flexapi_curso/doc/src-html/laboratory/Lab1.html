001    package laboratory;
002    
003    import java.net.URISyntaxException;
004    
005    import br.com.voicetechnology.flexapi.agentadapter.AgentAdapter;
006    
007    /**
008     * Laboratório 1.
009     * Informações mínimas necessárias para conexão com o servidor CTI.
010     *  
011     * @author Roberto Elvira
012     */
013    public class Lab1 {
014            private static final String RAMAL = "2150";
015            private static final int APP_NUMBER = 21501;
016            private static final String VERSION = "1.0.0";
017            private static final String URL = "//127.0.0.1:2556";
018            
019        private AgentAdapter api = null;
020        
021            public Lab1() throws URISyntaxException {
022            api = new AgentAdapter();
023            api.getApplicationInfo().getDevice().setRamal(RAMAL);
024            api.getApplicationInfo().setNumber(APP_NUMBER);
025            api.getApplicationInfo().setVersion(VERSION);
026            api.getStream().setURL(URL);
027            api.getStream().open();
028            }
029            public void close() {
030                    api.getStream().close();
031            }
032            
033            public static void main(String[] args) throws Exception {
034                    Lab1 main =     new Lab1();
035                    System.in.read();
036                    main.close();
037            }
038    }
