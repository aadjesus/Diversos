001    package laboratory;
002    
003    import java.net.URISyntaxException;
004    
005    import br.com.voicetechnology.flexapi.agentadapter.AgentAdapter;
006    import br.com.voicetechnology.flexapi.stream.FlexStreamListener;
007    import br.com.voicetechnology.util.Protocol;
008    
009    /**
010     * Laboratório 2.
011     * Verificação do estado da conexão com o servidor.
012     *  
013     * @author Roberto Elvira
014     */
015    
016    public class Lab2 implements FlexStreamListener {
017            private static final String RAMAL = "2150";
018            private static final int APP_NUMBER = 21501;
019            private static final String VERSION = "1.0.0";
020            private static final String URL = "//127.0.0.1:2556";
021            
022        private AgentAdapter api = null;
023        
024            public Lab2() throws URISyntaxException {
025            api = new AgentAdapter();
026            api.getApplicationInfo().getDevice().setRamal(RAMAL);
027            api.getApplicationInfo().setNumber(APP_NUMBER);
028            api.getApplicationInfo().setVersion(VERSION);
029            api.getStream().setURL(URL);
030            api.getStream().setReconnectTime(5);
031    
032            api.getStream().addListener(this);
033    
034            api.getStream().open();
035            }
036            public void close() {
037                    api.getStream().close();
038            }
039            public static void main(String[] args) throws Exception {
040                    Lab2 main =     new Lab2();
041                    System.in.read();
042                    main.close();
043            }
044    
045            public void onError() {
046                    System.out.println("Erro na conexao");
047            }
048            public void onDisconnect() {
049                    System.out.println("Desconectado do servidor");
050            }
051            public void onConnect() {
052                    System.out.println("Conectado ao servidor");
053            }
054            public void onReceive(Protocol arg0) {
055                    System.out.println("Recebeu dados:" + arg0.toString());
056            }
057    }
